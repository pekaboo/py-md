前端基础知识点汇总（JS/TS/CSS）
一、JavaScript 基础
1. 变量声明
var（函数作用域，存在变量提升）、let（块级作用域）、const（块级作用域，只读常量）
javascript
运行
// var（不推荐）
var a = 10;
var a = 20; // 允许重复声明
console.log(a); // 20

// let
let b = 10;
// let b = 20; // 报错：重复声明
if (true) {
  let b = 30;
}
console.log(b); // 10（块级作用域隔离）

// const（必须初始化，不可修改引用）
const c = 10;
// c = 20; // 报错：只读
const obj = { name: "张三" };
obj.name = "李四"; // 允许修改对象属性
2. 数据类型
基本类型：string、number、boolean、null、undefined、symbol、bigint
引用类型：object（包含数组、对象、函数等）
javascript
运行
// 基本类型
const str = "hello";
const num = 100;
const isTrue = false;
const n = null; // 表示空对象指针
const u = undefined; // 表示未定义
const sym = Symbol("唯一标识");
const big = 9007199254740991n; // 大整数

// 引用类型
const arr = [1, 2, 3];
const obj = { name: "张三", age: 20 };
const func = () => console.log("函数");

// 类型判断
console.log(typeof str); // "string"
console.log(typeof num); // "number"
console.log(typeof null); // "object"（历史bug）
console.log(Array.isArray(arr)); // true
console.log(obj instanceof Object); // true
3. 函数
普通函数、箭头函数（无this绑定、无arguments）
javascript
运行
// 普通函数
function add(a, b) {
  return a + b;
}
console.log(add(2, 3)); // 5

// 箭头函数
const multiply = (a, b) => a * b;
console.log(multiply(2, 3)); // 6

// 箭头函数与this
const person = {
  name: "张三",
  sayHi: function() {
    setTimeout(() => {
      console.log(`你好，${this.name}`); // this绑定person
    }, 1000);
  }
};
person.sayHi(); // 你好，张三
4. 数组常用方法
javascript
运行
const arr = [1, 2, 3, 4, 5];

// 遍历：forEach
arr.forEach(item => console.log(item)); // 1,2,3,4,5

// 映射：map（返回新数组）
const doubled = arr.map(item => item * 2);
console.log(doubled); // [2,4,6,8,10]

// 过滤：filter（返回满足条件的新数组）
const even = arr.filter(item => item % 2 === 0);
console.log(even); // [2,4]

// 汇总：reduce
const sum = arr.reduce((acc, curr) => acc + curr, 0);
console.log(sum); // 15
二、TypeScript 基础
TypeScript 是 JavaScript 的超集，核心是类型系统，编译后生成 JS。
1. 基本类型注解
typescript
运行
// 字符串
const str: string = "hello";

// 数字
const num: number = 100;

// 布尔值
const isTrue: boolean = false;

// 数组（两种写法）
const arr1: number[] = [1, 2, 3];
const arr2: Array<number> = [1, 2, 3];

// 元组（固定长度和类型）
const tuple: [string, number] = ["张三", 20];

// 枚举（默认从0开始，可自定义值）
enum Gender {
  Male,
  Female
}
const gender: Gender = Gender.Male;
console.log(gender); // 0

// 任意类型（不推荐，失去TS类型检查）
let anyVal: any = 10;
anyVal = "hello";

// 未知类型（安全的any，需类型断言后使用）
let unknownVal: unknown = 10;
// console.log(unknownVal.length); // 报错
if (typeof unknownVal === "string") {
  console.log((unknownVal as string).length); // 类型断言
}
2. 接口（Interface）
定义对象的结构契约
typescript
运行
interface Person {
  name: string; // 必选属性
  age?: number; // 可选属性
  readonly id: number; // 只读属性
}

const person: Person = {
  name: "张三",
  id: 1001
};
// person.id = 1002; // 报错：只读属性
person.age = 20; // 可选属性可赋值
3. 函数类型
typescript
运行
// 函数参数和返回值类型注解
function add(a: number, b: number): number {
  return a + b;
}

// 函数类型别名
type AddFunc = (a: number, b: number) => number;
const multiply: AddFunc = (a, b) => a * b;

// 可选参数（必须在必选参数后）
function greet(name: string, msg?: string): string {
  return msg ? `${msg}，${name}` : `你好，${name}`;
}
console.log(greet("张三")); // 你好，张三
4. 泛型（Generic）
实现类型复用，增强代码灵活性
typescript
运行
// 泛型函数：T为类型占位符，调用时指定具体类型
function identity<T>(arg: T): T {
  return arg;
}

const str = identity<string>("hello");
const num = identity<number>(100);

// 泛型接口
interface Result<T> {
  success: boolean;
  data: T;
}

const res: Result<Person> = {
  success: true,
  data: { name: "张三", id: 1001 }
};
三、CSS 基础
1. 选择器
css
/* 元素选择器 */
div {
  color: #333;
}

/* 类选择器（.开头） */
.container {
  width: 1200px;
  margin: 0 auto;
}

/* ID选择器（#开头，唯一） */
#header {
  height: 80px;
}

/* 后代选择器 */
.container p {
  line-height: 1.5;
}

/* 伪类选择器 */
a:hover {
  color: #f00;
}

/* 伪元素选择器 */
p::before {
  content: "前缀：";
}
2. 盒模型
css
.box {
  width: 200px; /* 内容宽度 */
  height: 200px; /* 内容高度 */
  padding: 20px; /* 内边距（上下左右） */
  border: 1px solid #000; /* 边框 */
  margin: 10px; /* 外边距（上下左右） */
  box-sizing: border-box; /* 盒模型：width/height 包含 padding+border（推荐） */
}
3. Flex 布局（弹性布局）
最常用的布局方式，用于快速实现对齐、分布等
css
/* 父容器 */
.flex-container {
  display: flex; /* 开启Flex布局 */
  flex-direction: row; /* 主轴方向：row（横向）/column（纵向） */
  justify-content: center; /* 主轴对齐：center（居中）/space-between（两端对齐）等 */
  align-items: center; /* 交叉轴对齐：center（居中）/stretch（拉伸）等 */
  gap: 20px; /* 子元素间距 */
  height: 400px;
}

/* 子元素 */
.flex-item {
  flex: 1; /* 占比（1表示平均分配剩余空间） */
  width: 100px;
  height: 100px;
  background: #f00;
}
4. 定位（Position）
css
/* static：默认值，无定位 */
.static {
  position: static;
}

/* relative：相对定位（相对于自身原位置） */
.relative {
  position: relative;
  top: 10px;
  left: 10px;
}

/* absolute：绝对定位（相对于最近的已定位祖先元素） */
.absolute {
  position: absolute;
  top: 50%;
  left: 50%;
  transform: translate(-50%, -50%); /* 水平垂直居中 */
}

/* fixed：固定定位（相对于浏览器窗口） */
.fixed {
  position: fixed;
  bottom: 20px;
  right: 20px;
}
5. 响应式布局（Media Query）
适配不同屏幕尺寸
css
/* 默认样式（大屏幕） */
.container {
  width: 1200px;
}

/* 平板（768px-1199px） */
@media (max-width: 1199px) {
  .container {
    width: 90%;
  }
}

/* 手机（≤767px） */
@media (max-width: 767px) {
  .container {
    width: 100%;
    padding: 0 10px;
  }
}
6. 动画
css
/* transition：过渡动画（适合简单状态变化） */
.btn {
  width: 100px;
  height: 40px;
  background: #00f;
  transition: all 0.3s ease; /* 所有属性变化过渡0.3秒，缓动效果 */
}
.btn:hover {
  background: #008000;
  transform: scale(1.1); /* 放大1.1倍 */
}

/* animation：关键帧动画（适合复杂动画） */
@keyframes rotate {
  0% { transform: rotate(0deg); }
  100% { transform: rotate(360deg); }
}
.loading {
  width: 50px;
  height: 50px;
  border: 5px solid #f00;
  animation: rotate 1s linear infinite; /* 1秒旋转一周，匀速，无限循环 */
}
